getMax :: [Int] -> Maybe Int
getMax [] = Nothing
getMax (x:xs) = case (getMax xs) of
  Nothing -> Just x
  Just y -> if x>y then Just x else Just y

reciprocal 0 = Nothing
reciprocal x = Just (1/x)
